# Họ và Tên Email Lớp / Khoá Điểm TB Trạng thái
📄 Xem code Ajax mẫu (nhấn để mở/đóng) ▾ 
// ── Bài Lab 19: Gọi Ajax với $.ajax ──────────────────────

var API_URL = 'https://67f5c0bc913986a7b5a3d6e7.mockapi.io/students';

$.ajax({
  url:      API_URL,
  method:   'GET',
  dataType: 'json',
  timeout:  8000,                    // ms – quá giờ → error

  // Trước khi gửi request
  beforeSend: function() {
    showLoading();               // hiện skeleton
  },

  // Thành công – data là mảng JSON
  success: function(data) {
    renderTable(data);
  },

  // Thất bại – xhr, status, error
  error: function(xhr, status, err) {
    showError(status + ' – ' + err);
  },

  // Luôn chạy dù success hay error
  complete: function() {
    hideLoading();
  }
});

// ── Hoặc dùng cú pháp $.get ngắn gọn hơn ────────────────

$.get(API_URL)
  .done(function(data)  { renderTable(data); })
  .fail(function(xhr)   { showError(xhr.statusText); })
  .always(function()    { hideLoading(); });
